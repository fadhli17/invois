# filepath: /Applications/XAMPP/xamppfiles/htdocs/mza/.htaccess

RewriteEngine On
RewriteBase /mza/

# Halang direct directory access
Options -Indexes

# Benarkan akses ke fail PHP
<FilesMatch "\.php$">
    Order Allow,Deny
    Allow from all
</FilesMatch>

# Lindungi fail sensitif
<Files .htaccess>
    Order allow,deny
    Deny from all
</Files>

<Files .env>
    Order allow,deny
    Deny from all
</Files>

# Redirect dari .php extension ke clean URL
RewriteCond %{THE_REQUEST} ^[A-Z]{3,}\s([^.]+)\.php [NC]
RewriteRule ^ %1 [R=301,L]

# Handle clean URLs
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME}.php -f
RewriteRule ^(.*)$ $1.php [L]

# Benarkan POST requests
RewriteCond %{REQUEST_METHOD} ^(POST)
RewriteRule ^ - [L]
